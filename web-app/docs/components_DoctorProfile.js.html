import React from 'react';
import {
    Container,
    Grid,
    List,
    ListItem,
    Typography,
    Paper,
    CircularProgress,
    Backdrop,
    ListItemText,
} from '@material-ui/core'
import { makeStyles } from '@material-ui/core/styles'
import { useAuth } from '../contexts/AuthContext'

const useStyles = makeStyles(theme => ({
    backdrop: {
        zIndex: theme.zIndex.drawer + 1,
        color: '#fff',
    },
}))

/**
 * @description Doctor profile page.
 * @version 1.0.1
 * @name Profile
 * @author [Marin Jereghi](https://github.com/marinjereghi)
 * @param {*} props 
 * @returns 
 */
export function Profile(props) {
    const { location, history } = props;
    const [state, setState] = React.useState({
        patients: [

        ],
        backdropOpen: false,
    })
    const classes = useStyles();
    const { currentUser, getPatients } = useAuth();
    const isMounted = React.useRef(null);
    React.useEffect(() => {
        const fetchData = async () => {


            try {

                if (isMounted.current)
                    setState(s => ({ ...s, backdropOpen: true }))

                const snapshot = await getPatients();
                const patients = snapshot.docs.map(doc => ({ ...doc.data(), id: doc.id }))


                if (isMounted.current)
                    setState(s => ({ ...s, backdropOpen: false, patients: patients }))

            } catch (error) {
                if (isMounted.current)
                    setState(s => ({ ...s, backdropOpen: false }))
            }

        }
        isMounted.current = true
        fetchData();
        return () => (isMounted.current = false)
    }, [])

    const { backdropOpen } = state;

    if (backdropOpen) {
        return (
            <Backdrop className={classes.backdrop} open={backdropOpen}>
                <CircularProgress color="inherit" />
            </Backdrop>
        )
    }

    return (
        <Container>
            <Grid container spacing={3} direction="row" justifyContent="flex-start" alignItems="flex-start">
                <Grid item xs={4}>
                    <Paper elevation={3} variant="elevation" style={{ padding: 12 }}>
                        <Typography variant="body1" color="textPrimary">{`Doctor ${currentUser.displayName}`}</Typography>
                        <Typography variant="body2" color="textSecondary">{`${currentUser.email}`}</Typography>

                    </Paper>
                </Grid>
                <Grid item xs={4}>
                    <Paper elevation={3} variant="elevation" style={{ padding: 12 }}>
                        {"altri dati"}
                    </Paper>
                </Grid>
                <Grid item xs={4} >
                    <Paper elevation={3} variant="elevation" style={{ padding: 12 }}>
                        <Typography variant="h6" gutterBottom color="textSecondary">Lista pazienti:</Typography>
                        {state.patients && <List>
                            {state.patients.map(item => (
                                <ListItem key={item.id}>
                                    <ListItemText primary={`${item.name} ${item.surname}`} secondary={item.email} />
                                </ListItem>
                            ))}
                        </List>}
                    </Paper>
                </Grid>
            </Grid>
        </Container>
    )
}
