import React from "react"
import { BrowserRouter as Router, Route, Switch, Redirect } from "react-router-dom"
import { AuthProvider } from "./contexts/AuthContext"
import PageNotFound from "./components/PageNotFound"
import SecureRoute from "./components/SecureRoute"
import AppLayout from "./layouts/AppLayout"
import AuthLayout from "./layouts/AuthLayout"
import detectBrowserLanguage from 'detect-browser-language'
import Language from "./components/Language"

/**
 * Root application component.
 * @version 1.0.1
 * @name App
 * @author [Marin Jereghi](https://github.com/marinjereghi)
 * @returns Router component
 */
function App() {

    Language.setLanguage(detectBrowserLanguage());

    return (
        <Router>
            <AuthProvider>
                <Switch>
                    <SecureRoute path="/reserved" component={AuthLayout} />
                    {/* */}
                    <Route path="/signup" component={AppLayout} />
                    <Route path="/login" component={AppLayout} />
                    <Route path="/forgot-password" component={AppLayout} />
                    {/* <Route path="/forgot-password" component={ForgotPassword} /> */}
                    <Redirect from="/" to="/reserved/" />
                    <Route path="*" component={PageNotFound} />
                </Switch>
            </AuthProvider>
        </Router>
    )
}

export default App;
